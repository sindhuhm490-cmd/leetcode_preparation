1929. Concatenation of Array
      -----------------------
  class Solution {
    public int[] getConcatenation(int[] nums) {
        int n=nums.length;
        int [] ans=new int[2*n];
       
        for(int i=0;i<n;i++){
            ans[i]=nums[i] ;
            ans[i+n]=ans[i];
            //ans1[i]=ans[i]+nums[i];

        }
        
        
       return ans; 

    }
}
